\subsubsection{Round, CAF, and Capacity Parameters}
\begin{itemize}
    \item $\text{nominal\_week}(r)$: Ideal calendar week for round $r$, computed by linear interpolation.
    \item $\text{caf\_dates}(t)$: Sorted list of CAF match dates for team $t \in C$.
    \item $\text{CAF\_BUFFER}$: Minimum calendar days apart from any CAF date $= 4$.
    \item $MAX\_DAY = 3$: Max league matches on one calendar date.
    \item $MAX\_SLOT = 2$: Max simultaneous matches in the same kickoff slot.
    \item $HARD\_MAX\_WEEK = 18$: Hard upper bound on matches per calendar week.
    \item $SOFT\_MIN\_WEEK = 6$: Soft lower target for matches per calendar week.
    \item $SOFT\_MAX\_WEEK = 12$: Soft upper target for matches per calendar week.
    \item $MIN\_REST\_DAYS\_LOCAL = 3$: Min calendar days between consecutive league matches.
    \item $MAX\_STREAK = 2$: Max consecutive home (or away) matches.
\end{itemize}

\subsection{Feasible Domain}
Let $\text{feasible\_domain}(m)$ be the subset of $S$ that match $m$ is permitted to use before the solver runs. A slot $s \in \text{feasible\_domain}(m)$ if and only if:
\begin{enumerate}
    \item $\text{date}(s) \notin \text{FIFA\_DATES}$
    \item $\text{date}(s)$ has valid date, week, and kickoff metadata
    \item For each $t \in \text{team}(m) \cap C$, and for each $d \in \text{caf\_dates}(t)$:
    $|\text{date}(s) - d| \ge \text{CAF\_BUFFER}$
\end{enumerate}

\subsection{Decision Variables}
\textbf{Primary assignment:}
\begin{equation}
x_{m,s,v} \in \{0, 1\}, \quad \forall m \in M, \ s \in \text{feasible\_domain}(m), \ v \in V
\end{equation}
$x_{m,s,v} = 1$ if and only if match $m$ is assigned to calendar slot $s$ at venue $v$. Let $x_{m,s} = \sum_{v \in V} x_{m,s,v}$.

\textbf{Auxiliary Variables:}
\begin{align}
\text{match\_time}_m &\in \mathbb{Z}^+, \quad \forall m \in M \\
\text{round\_min}_r, \text{round\_max}_r &\in \mathbb{Z}^+, \quad \forall r \in R \\
\text{load}_w, \text{under\_load}_w, \text{over\_load}_w &\in \mathbb{Z}^+, \quad \forall w \in W \\
\text{fr\_slots}_s &\in \{0,1\} \quad \text{for final-round slots} \\
\text{slot\_collisions}_s &\in \{0,1\} \\
\text{overlap}_{v,d} &\in \{0,1\}
\end{align}

\subsection{Hard Constraints}

\textbf{H1 - Assignment uniqueness:}
\begin{equation}
\sum_{s \in \text{feasible\_domain}(m)} x_{m,s} = 1 \quad \forall m \in M
\end{equation}

\textbf{H2 - Tier-1 derby slot requirement:}
\begin{equation}
\sum_{s: \text{tier\_slot}(s)=1} x_{m,s} = 1 \quad \forall m : \text{is\_derby}(m)=1
\end{equation}

\textbf{H3 - Team date uniqueness:}
\begin{equation}
\sum_{m: t \in \text{team}(m)} \sum_{s: \text{date}(s)=d} x_{m,s} \le 1 \quad \forall t \in T, d \in D
\end{equation}

\textbf{H4 - Venue slot uniqueness:}
\begin{equation}
\sum_{m \in M} x_{m,s,v} \le 1 \quad \forall v \in V, s \in S
\end{equation}

\textbf{H5 - Daily match cap:}
\begin{equation}
\sum_{m} \sum_{s: \text{date}(s)=d} x_{m,s} \le MAX\_DAY \quad \forall d \in D
\end{equation}

\textbf{H6 - Slot concurrency cap:}
\begin{equation}
\sum_{m} x_{m,s} \le MAX\_SLOT \quad \forall s \in S
\end{equation}

\textbf{H7 - Team rest (sliding window):}
\begin{equation}
\sum_{m: t \in \text{team}(m)} \sum_{s: \text{date}(s) \in [d, d+MIN\_REST\_DAYS\_LOCAL-1]} x_{m,s} \le 1 \quad \forall t \in T, d \in D
\end{equation}

\textbf{H8 - Match time definition:}
\begin{equation}
\text{match\_time}_m = \sum_{s} \text{time\_index}(s) \cdot x_{m,s} \quad \forall m \in M
\end{equation}

\textbf{H9 - Round bounds definition:}
\begin{align}
\text{round\_min}_r &= \min_{m: \text{round}(m)=r} \text{match\_time}_m \quad \forall r \in R \\
\text{round\_max}_r &= \max_{m: \text{round}(m)=r} \text{match\_time}_m \quad \forall r \in R
\end{align}

\textbf{H10 - Strict round ordering:}
\begin{equation}
\text{round\_max}_r + MIN\_ROUND\_GAP \le \text{round\_min}_{r+1} \quad \forall r \in \{1, 2, \dots, 33\}
\end{equation}

\textbf{H11 - Weekly hard cap:}
\begin{equation}
\sum_{m} \sum_{s: \text{week}(s)=w} x_{m,s} \le HARD\_MAX\_WEEK \quad \forall w \in W
\end{equation}

\textbf{H14 \& H15 - Final-Round Rule:}
\begin{equation}
\sum_{s \in FR\_SLOTS} \text{fr\_slots}_s = 1
\end{equation}
\begin{equation}
\sum_{v} x_{m,s,v} = \text{fr\_slots}_s \quad \forall m: \text{round}(m)=34
\end{equation}

\subsection{Objective Function}
Minimize $Z = Z_1 + Z_2 + Z_3 + Z_4 + Z_5 + Z_6 + Z_7 + Z_8$

\begin{align}
Z_1 &= W\_ROUND\_ORDER \cdot \sum_{m} \sum_{s} |\text{week}(s) - \text{nominal\_week}(\text{round}(m))| \cdot x_{m,s} \\
Z_2 &= W\_WEEK\_UNDERLOAD \cdot \sum_{w} \text{under\_load}_w \\
Z_3 &= W\_WEEK\_OVERLOAD \cdot \sum_{w} \text{over\_load}_w \\
Z_4 &= W\_TRAVEL \cdot \sum_{m} \sum_{s} \text{travel\_km}(m) \cdot x_{m,s} \\
Z_5 &= W\_TIER\_MISMATCH \cdot \sum_{m} \sum_{s} |\text{tier\_match}(m) - \text{tier\_slot}(s)| \cdot x_{m,s} \\
Z_6 &= W\_EVENING\_PREFERENCE \cdot \sum_{m} \sum_{s} \max(0, 21 - \text{kickoff\_hour}(s)) \cdot x_{m,s} \\
Z_7 &= W\_SLOT\_SPREAD \cdot \sum_{s} \text{slot\_collisions}_s \\
Z_8 &= W\_STADIUM\_MAINTENANCE\_OVERLAP \cdot \sum_{(v,d)} \text{overlap}_{v,d}
\end{align}
